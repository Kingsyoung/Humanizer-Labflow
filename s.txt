class SentenceData(BaseModel):
    id: str
    original: str
    humanized: str
    alternatives: List[str]
    score: float

def protect_citations(text):
    citation_patterns = [
        r'\([A-Z][a-z]+(?:\s+et\s+al\.)?,\s*\d{4}[a-z]?\)',
        r'\[[\d,\s\-]+\]',
        r'\([A-Z]{2,}(?:\s+et\s+al\.)?,\s*\d{4}\)',
        r'[A-Z][a-z]+\s*\(\d{4}\)',
    ]
    protected_text = text
    citations = []
    for pattern in citation_patterns:
        matches = re.finditer(pattern, protected_text)
        for match in matches:
            placeholder = f"__CITE_{len(citations)}__"
            citations.append(match.group())
            protected_text = protected_text.replace(match.group(), placeholder, 1)
    return protected_text, citations

def restore_citations(text, citations):
    for i, cite in enumerate(citations):
        text = text.replace(f"__CITE_{i}__", cite)
    return text

def split_sentences(text):
    protected_text, citations = protect_citations(text)
    protected_text = re.sub(r"\b(e\.g\.|i\.e\.|et al\.|Fig\.|Dr\.|Prof\.)\s", lambda m: m.group(0).replace(".", "\x00"), protected_text)
    sents = re.split(r"(?<=[.!?])\s+(?=[A-Z])", protected_text.strip())
    sents = [s.replace("\x00", ".").strip() for s in sents if s.strip()]
    final_sents = []
    for sent in sents:
        sent = restore_citations(sent, citations)
        final_sents.append(sent)
    return final_sents

def split_paragraphs(text):
    return [split_sentences(p.strip()) for p in text.split("\n\n") if p.strip()]

def score_sentence(sent):
    s, words = sent.lower(), sent.split()
    score = sum(15 for t in AI_TELLS if t in s)
    
    # Length uniformity penalty
    if 15 <= len(words) <= 22: score += 15
    elif 12 <= len(words) <= 25: score += 8
    
    # Transition starter penalty
    first = words[0].lower().strip(",.:") if words else ""
    transition_starters = ["furthermore", "moreover", "however", "therefore", "thus", 
                          "consequently", "additionally", "crucially", "notably", "significantly"]
    if first in transition_starters: score += 15
    
    # Vocabulary diversity penalty (high diversity = predictable)
    unique = len(set(w.lower() for w in words))
    if len(words) > 5:
        diversity = unique / len(words)
        if diversity > 0.75: score += 10  # Too diverse = AI-like
        if diversity < 0.45: score += 8   # Too repetitive
    
    # Passive voice density
    passive = ["is", "are", "was", "were", "be", "been", "being"]
    passive_count = sum(1 for w in words if w.lower() in passive)
    if passive_count >= 2 and len(words) < 30: score += 8
    
    # Exact phrase matches
    for phrase in ["furthermore", "moreover", "in conclusion", "it is important to note", 
                   "it is crucial to note", "crucially", "notably"]:
        if phrase in s: score += 20
    
    return min(100, max(0, score))

def humanize_with_mistral(paragraphs, style):
    print(f"CALLING MISTRAL with {len(paragraphs)} paragraphs")
    lines = []
    for i, para in enumerate(paragraphs):
        lines.append(f"Paragraph {i+1}:")
        for j, s in enumerate(para):
            lines.append(f"{j+1}. {s}")
        lines.append("")
    
    resp = client.chat.complete(
        model="mistral-large-latest",
        messages=[
            {"role": "system", "content": SYSTEM},
            {"role": "user", "content": f"Style: {style}\n\nHumanize this academic text:\n\n" + "\n".join(lines)}
        ],
        temperature=0.9,
        max_tokens=4000,
        response_format={"type": "json_object"}
    )
    
    text = resp.choices[0].message.content
    print(f"RAW RESPONSE: {text[:200]}...")
    data = json.loads(text.strip())
    result = []
    
    for i, para in enumerate(data.get("processed_paragraphs", [])):
        sents = []
        for j, sent in enumerate(para.get("sentences", [])):
            h = sent.get("humanized", "")
            
            # NUCLEAR OPTION: Aggressive mathematical anti-detection
            h = aggressive_anti_detection(h)
            
            score = score_sentence(h)
            print(f"  SENTENCE {j}: initial score {score}")
            
            # Multiple critic passes if needed
            attempts = 0
            while score > 25 and attempts < 3:
                print(f"    CRITIC PASS {attempts+1}: score {score}")
                
                # Surgical feedback based on specific issues
                issues = []
                words = h.split()
                if 12 <= len(words) <= 25:
                    issues.append("length too uniform")
                if words[0].lower() in ["the", "this", "these", "that"]:
                    issues.append("predictable start")
                if len(set(w.lower() for w in words)) / len(words) > 0.7:
                    issues.append("vocab too diverse")
                if ";" not in h and "—" not in h:
                    issues.append("no punctuation variety")
                
                feedback = f"Rewrite. Issues: {', '.join(issues)}. "
                feedback += "Make it shorter or longer. Add a fragment. Use 'we' or 'but'. Keep citations exact."
                
                critic = client.chat.complete(
                    model="mistral-large-latest",
                    messages=[
                        {"role": "system", "content": SYSTEM},
                        {"role": "user", "content": feedback + "\n\nSentence: " + h}
                    ],
                    temperature=0.95,
                    max_tokens=500
                )

            sents.append(SentenceData(
                id=f"p{i}-s{j}",
                original=sent.get("original", ""),
                humanized=h,
                alternatives=sent.get("alternatives", [])[:3],
                score=score
            ))
        result.append(ParagraphData(id=f"para-{i}", sentences=sents))
    print(f"RETURNING {len(result)} paragraphs")
    return result

@app.post("/api/process-text", response_model=ProcessResponse)
async def process(req: ProcessRequest):
    if not req.text.strip():
        raise HTTPException(400, "Empty text")
    pars = split_paragraphs(req.text)
    processed = humanize_with_mistral(pars, req.style)
    all_s = [s for p in processed for s in p.sentences]
    avg = sum(s.score for s in all_s) / len(all_s) if all_s else 0
    return ProcessResponse(
        processed_paragraphs=processed,
        total_sentences=len(all_s),
        avg_score=round(avg, 1)
    )
